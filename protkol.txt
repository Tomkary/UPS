//connect to the server
connect|string(nick)\n
    connect|ok\n
    connect|err(id)\n

//connect to the room
join|string(room id)\n
    join|ok|int(player id)\n
    join|err(id)\n

//crate a new room
croom|string(name)\n
	croom|ok\n
		lobby
	croom|err(id)\n

//leave room
leave\n
    nic

//pause the game 
pause|start|int(player id)\n
    nic

//end pause
pause|end|int(player id)\n
    status

    turn|ok|string[](karty)\n

    turn|err(id)\n
